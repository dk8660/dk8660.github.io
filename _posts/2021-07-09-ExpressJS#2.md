# Home Page 구현

```javascript
const express = require('express')
const app = express()
const port = 3000
let fs = require('fs')
let template = require('./lib/template.js')

app.get('/', (request, response) => {
  fs.readdir('./data', function(error, filelist){
    var title = 'Welcome';
    var description = 'Hello, Node.js';
    var list = template.list(filelist);
    var html = template.HTML(title, list,
      `<h2>${title}</h2>${description}`,
      `<a href="/create">create</a>`
    );
    response.send(html)
  });
})

app.listen(port, () => {
  console.log(`Example app listening at http://localhost:${port}`)
})
```

# Page Create 구현

```javascript
app.get('/create', function(request, response) {
  fs.readdir('./data', function(error, filelist){
    var title = 'WEB - create';
    var list = template.list(filelist);
    var html = template.HTML(title, list, `
      <form action="/create_process" method="post">
        <p><input type="text" name="title" placeholder="title"></p>
        <p>
          <textarea name="description" placeholder="description"></textarea>
        </p>
        <p>
          <input type="submit">
        </p>
      </form>
    `, '');
    response.send(html);
  });
})

```

Create 페이지를 만드는 것은 위에서 다루었던 다른 페이지를 만드는 것과 다르지 않다.

Routing을 이용하여 접근한다.



`<form action="/create_process" method="post">`

이 부분에서 데이터 전송 방식을 보면 post 형식이다.

따라서 실제로 페이지를 만드려면 app.post() 함수가 필요하다.

```javascript
app.post('/create_process', function(request, response) {
  var body = '';
  request.on('data', function(data){
      body = body + data;
  });
  request.on('end', function(){
      var post = qs.parse(body);
      var title = post.title;
      var description = post.description;
      fs.writeFile(`data/${title}`, description, 'utf8', function(err){
        response.writeHead(302, {Location: `/?id=${title}`});
        response.end();
      })
  });
})
```

# Page Delete 구현

```javascript
app.post('/delete_process', function(request, response) {
  var body = '';
  request.on('data', function(data){
      body = body + data;
  });
  request.on('end', function(){
      var post = qs.parse(body);
      var id = post.id;
      var filteredId = path.parse(id).base;
      fs.unlink(`data/${filteredId}`, function(error){
        response.writeHead(302, {Location: `/`});
        response.end();
      })
  });
}) 
```

delete 페이지는 따로 만들 필요가 없기에 우리는 delete가 이루어지는 과정만을 만들어주면 된다.

따라서 app.post()만을 이용하여 구현해준다.



```javascript
response.writeHead(302, {Location: `/`});
    response.end();
```

 ExpressJS는 Redirect를 편리하게 할 수 있는 기능을 갖고 있어 위 코드를 다음과 같이 짧게 표현할 수 있다.

```javascript
response.redirect('/')
```

위와 같은 방식으로 다른 부분의 Redirect도 바꿔주면 된다.
